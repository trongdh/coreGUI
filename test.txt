// Script to parse drop-down value
window.onload = function () {
  var countrySelect = document.getElementById("countrySelect");
  var templateSelect = document.getElementById("templateSelect");
  var periodSelect = document.getElementById("periodSelect");

  function populateDropdown(selectElement, options) {
    selectElement.length = 1;
    for (var option in options) {
      selectElement.options[selectElement.options.length] = new Option(option, option);
    }
  }

  countrySelect.onchange = function () {
    periodSelect.length = 1;
    templateSelect.length = 1;
    populateDropdown(templateSelect, subjectObject[this.value]);
  };

  templateSelect.onchange = function () {
    periodSelect.length = 1;
    var z = subjectObject[countrySelect.value][this.value];
    for (var i = 0; i < z.length; i++) {
      periodSelect.options[periodSelect.options.length] = new Option(this.value + "_" + z[i], this.value + "_" + z[i]);
    }
  };
};
